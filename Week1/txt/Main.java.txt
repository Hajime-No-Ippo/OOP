import java.io.File;

public class Main {
	
	public static void main(String args[]) {
		File root = new File("/Users/tobymiles/CS627C OOP/Week1");
		FileSystemNode tree = FileExplorer.buildTree(root);
		tree.print("");
		
		if(tree instanceof DirectoryNode) {

		}
		
	}

}
